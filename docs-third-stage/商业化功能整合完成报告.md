
### 2. 更新 members 集合

```javascript
// 为现有成员添加积分字段
const members = await db.collection('members').get();

for (const member of members.data) {
  if (!member.starlightPoints) {
    await db.collection('members').doc(member._id).update({
      data: {
        starlightPoints: 0,
        crystalDiamonds: 0,
        totalStarlight: 0,
        totalCrystal: 0,
        personalExp: 0,
        level: 1
      }
    });
  }
}
```

## 📝 技术细节

### 事务处理

```javascript
const transaction = await db.startTransaction();

try {
  // 1. 扣减积分
  await transaction.collection('members').doc(userId).update({
    data: { starlightPoints: _.inc(-pointsCost) }
  });
  
  // 2. 创建优惠券
  await transaction.collection('couponRedemptions').add({
    data: couponData
  });
  
  await transaction.commit();
} catch (error) {
  await transaction.rollback();
  throw error;
}
```

### 防刷机制

```javascript
// 检查本月已兑换数量
const startOfMonth = new Date();
startOfMonth.setDate(1);
startOfMonth.setHours(0, 0, 0, 0);

const count = await db.collection('couponRedemptions')
  .where({
    userId: OPENID,
    createdAt: _.gte(startOfMonth.getTime()),
    status: _.in(['unused', 'used'])
  })
  .count();

if (count.total >= 2) {
  return { success: false, message: '已达月上限' };
}
```
